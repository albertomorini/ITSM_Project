
\newglossaryentry{gkpis}{
    name={Key Performance Indicator},
    text=g,
    description={in IT Service Management is a quantifiable metric used to evaluate how effectively IT services and processes are achieving defined business objectives and outcomes}
}
\newglossaryentry{gcsf}{
    name={Critical successful Factor},
    description={is a key area of activity that must be performed successfully for an organization, service, or process to achieve its strategic objectives}
}
\newglossaryentry{gitsm}{
    name={IT Service Management},
    description={a set of organizational capabilities, practices, and processes for designing, delivering, managing, and continually improving IT services to meet business and user needs.}
}

\newglossaryentry{gpestle}{
    name={PESTLE},
    description={A strategic analysis framework used to identify external factors affecting an organization. Political: government policies and regulations; Economic: inflation, costs, and market conditions; Social: demographics and customer expectations; Technological: innovation and emerging technologies; Legal: laws and compliance requirements; Environmental: sustainability and environmental regulations}
}
\newglossaryentry{gsla}{
    name={Service Level Agreements},
    description={A documented agreement between a service provider and a customer, written in natural langueage, that identifies both the services required and the expected level of service}
}
\newglossaryentry{gci}{
    name={Configuration Item},
    description={Any component that needs to be managed in order to deliver an IT service. A CI may include hardware, software, documentation, databases, network devices, applications, services, or any other asset whose configuration and relationships are controlled through Service Configuration Management}
}
